template <typename T>
void shakerSort (T* A, size_t n) {
int low = 0, up = n - 1;
int last;
T tmp;
while (low < up) {
last = -1;
for (int i = low; i < up; i++) {
if (A[i] > A[i + 1]) {
tmp = A[i];
A[i] = A[i + 1];
A[i + 1] = tmp;
last = i;
}
}
up = last;
last = n;
for (int i = up - 1; i >= low; i--) {
if (A[i] > A[i + 1]) {
tmp = A[i];
A[i] = A[i + 1];
A[i + 1] = tmp;
last = i;
}
}
low = last + 1;
}
}
